233 Gradient Gx(k, m, dx);
234 Gradient Gy(k, n, dy);
235
236 sp_mat Im = speye(m + 2, m + 2);
237 sp_mat In = speye(n + 2, n + 2);
238
239 Im.shed_row(0);
240 Im.shed_row(m);
241 In.shed_row(0);
242 In.shed_row(n);
243
244 sp_mat G1 = Utils::spkron(In, Gx);
245 sp_mat G2 = Utils::spkron(Gy, Im);
246
247 // Dimensions = 2*m*n+m+n, (m+2)*(n+2)
248 if (m != n)
249 *this = Utils::spjoin_cols(G1, G2);
250 else {
251 sp_mat A1(2, 1);
252 sp_mat A2(2, 1);
253 A1(0, 0) = A2(1, 0) = 1.0;
254 *this = Utils::spkron(A1, G1) + Utils::spkron(A2, G2);
255 }

267

273 Io.shed_row(o);

275 sp_mat G1 = Utils::spkron(Utils::spkron(Io, In), Gx);
276 sp_mat G2 = Utils::spkron(Utils::spkron(Io, Gy), Im);
277 sp_mat G3 = Utils::spkron(Utils::spkron(Gz, In), Im);
278
279 // Dimensions = 3*m*n*o+m*n+m*o+n*o, (m+2)*(n+2)*(o+2)
280 if ((m != n) || (n != o))
281 *this = Utils::spjoin_cols(Utils::spjoin_cols(G1, G2), G3);
282 else {
283 sp_mat A1(3, 1);
284 sp_mat A2(3, 1);
285 sp_mat A3(3, 1);
286 A1(0, 0) = A2(1, 0) = A3(2, 0) = 1.0;
287 *this =
288 Utils::spkron(A1, G1) + Utils::spkron(A2, G2) + Utils::spkron(A3, G3);
289 }
